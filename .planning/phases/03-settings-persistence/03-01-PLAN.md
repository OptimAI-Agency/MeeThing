import { useQuery } from "@tanstack/react-query";
import { supabase } from "@/integrations/supabase/client";
import { useAuth } from "@/contexts/AuthContext";

export function useCalendarConnections() {
  const { user } = useAuth();
  return useQuery({
    queryKey: ["calendar-connections", user?.id],
    enabled: !!user,
    queryFn: async () => {
      const { data, error } = await supabase
        .from("calendar_connections")
        .select("id, provider, connected_at, last_synced_at, is_active")
        .eq("is_active", true)
        .order("connected_at", { ascending: true });
      if (error) throw error;
      return data;
    },
  });
}
```

-- Phase 3: Add wellness columns to user_settings (D-09)
-- These columns are persisted by Phase 3 and read by Phase 4 (D-10)

ALTER TABLE public.user_settings
  ADD COLUMN wellness_tips_enabled BOOLEAN NOT NULL DEFAULT true,
  ADD COLUMN auto_breaks_enabled BOOLEAN NOT NULL DEFAULT false,
  ADD COLUMN breathing_reminder_enabled BOOLEAN NOT NULL DEFAULT false,
  ADD COLUMN breathing_reminder_minutes INTEGER NOT NULL DEFAULT 10,
  ADD COLUMN transition_buffer_enabled BOOLEAN NOT NULL DEFAULT false;

-- Fix CHECK constraint on reminder_minutes: current constraint allows (0, 5, 15, 30, 60)
-- but UI offers value "10" which is NOT in the constraint (Pitfall 4)
ALTER TABLE public.user_settings
  DROP CONSTRAINT IF EXISTS user_settings_reminder_minutes_check;
ALTER TABLE public.user_settings
  ADD CONSTRAINT user_settings_reminder_minutes_check
  CHECK (reminder_minutes IN (0, 5, 10, 15, 30, 60));

-- Add CHECK constraint for breathing_reminder_minutes (valid: 5, 10, 15)
ALTER TABLE public.user_settings
  ADD CONSTRAINT user_settings_breathing_reminder_minutes_check
  CHECK (breathing_reminder_minutes IN (5, 10, 15));

-- Add INSERT RLS policy so upsert works as safety net (Pitfall 1 from RESEARCH.md)
CREATE POLICY "Users can insert their own settings"
  ON public.user_settings FOR INSERT
  WITH CHECK (auth.uid() = user_id);
```

Create `src/hooks/useUserSettings.ts` following the `useCalendarConnections` pattern (per D-01, D-02, D-03). The hook must:

1. **Export a named function** `useUserSettings` (not default export -- matches `useCalendarConnections` pattern).

2. **Read path** using `useQuery`:
   - Query key: `["user-settings", user?.id]`
   - Enabled only when `!!user`
   - Select these columns from `user_settings`: `sync_frequency_minutes, notifications_enabled, reminder_minutes, wellness_tips_enabled, auto_breaks_enabled, breathing_reminder_enabled, breathing_reminder_minutes, transition_buffer_enabled`
   - Filter: `.eq("user_id", user!.id)`
   - Use `.single()` (one row per user guaranteed by schema -- Pitfall 5)
   - Throw on error

3. **Write path** using `useMutation`:
   - `mutationFn` accepts a payload object with fields: `sync_frequency_minutes: number`, `notifications_enabled: boolean`, `reminder_minutes: number`, `wellness_tips_enabled: boolean`, `auto_breaks_enabled: boolean`
   - Uses `supabase.from("user_settings").update(payload).eq("user_id", user!.id)` (per RESEARCH recommendation -- `.update()` not `.upsert()` since trigger guarantees row exists, but INSERT policy is added as safety net)
   - Throws on error
   - `onSuccess`: calls `queryClient.invalidateQueries({ queryKey: ["user-settings"] })`
   - Does NOT handle toasts (toasts are handled by the caller in CalendarSettings)

4. **Return shape**: `{ data, isLoading, isError, error, refetch, saveSettings }` where `saveSettings` is the mutation object (caller uses `saveSettings.mutate(payload)` and `saveSettings.isPending`).

5. **Field mapping helpers** (exported for use by CalendarSettings):
   - `mapDbToUi(db)`: Converts DB row to UI state shape:
     - `syncFrequency: String(db.sync_frequency_minutes)` (per D-12)
     - `notifications: db.notifications_enabled` (per D-13)
     - `reminderTime: String(db.reminder_minutes)` (per D-14)
     - `wellnessTips: db.wellness_tips_enabled` (per D-15)
     - `autoBreaks: db.auto_breaks_enabled` (per D-16)
   - `mapUiToDb(ui)`: Converts UI state to DB payload:
     - `sync_frequency_minutes: parseInt(ui.syncFrequency, 10)` (per D-12)
     - `notifications_enabled: ui.notifications` (per D-13)
