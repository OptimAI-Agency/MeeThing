```typescript
const [draft, setDraft] = useState<UiSettings>(defaultSettings);

  if (dbSettings) {
    setDraft(mapDbToUi(dbSettings));
  }
}, [dbSettings]);
```

3. `updateSetting` function (same signature as current, updates draft):
```typescript
const updateSetting = (key: keyof UiSettings, value: string | boolean) => {
  setDraft(prev => ({ ...prev, [key]: value }));
};
```

4. `handleSave` function (per D-04 explicit save, D-05 success toast, D-06 error toast):
```typescript
const handleSave = () => {
  saveSettings.mutate(mapUiToDb(draft), {
    onSuccess: () => {
      toast({
        variant: "success",
        title: "Settings Saved",
        description: "Your calendar preferences have been updated successfully.",
      });
    },
    onError: () => {
      toast({
        variant: "destructive",
        title: "Save Failed",
        description: "Could not save your settings. Please try again.",
      });
    },
  });
};
```

5. **Loading state (per D-07):** Wrap the settings panels (`SyncSettings` + `NotificationSettings`) in a div that applies `opacity-50 pointer-events-none` when `isLoading` is true:

```typescript
<div className={isLoading ? "opacity-50 pointer-events-none" : ""}>

</div>
```

6. **Error state (per D-08):** Before the settings panels, add an inline error banner when `isError` is true:
```typescript
{isError && (
  <div className="glass-panel rounded-3xl p-6 text-center space-y-3">
    <p className="text-red-600 font-medium">Could not load your settings</p>
    <Button variant="outline" onClick={() => refetch()} className="rounded-xl">
      Try Again
    </Button>
  </div>
)}
```

7. **Save button:** Disable when `saveSettings.isPending` or `isLoading`. Show "Saving..." text when `isPending`:
```typescript
<Button
  onClick={handleSave}
  disabled={saveSettings.isPending || isLoading}

```
